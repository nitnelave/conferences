\begin{lrbox}{\codebox}
  \begin{tcblisting}{listing only,title=Menhir: Basic arithmetics}
E:
| exp=E `+' term=T     { exp + term }
| exp=E `-' term=T     { exp - term }
| term=T               { term       }

T:
| term=T `*' factor=F  { term * factor }
| num=T `/' denom=F    { num / denom   }
| factor=F             { factor        }

F:
| n=NUMBER             { n    }
| `-' neg=F            { -neg }
| `(' exp=E `)'        { exp  }
  \end{tcblisting}
\end{lrbox}

\begin{lrbox}{\codebox}
  \begin{tcolorbox}[title=\centering{Parametrized rules}\\Bison\hspace{0.45\linewidth} Menhir,sidebyside]%
  \begin{lstlisting}
enum:
  ENUM NAME `{' value_list `}'
  { $$ = new EnumNode($2, $4); }
| ENUM NAME `;'
  { $$ = new EnumNode($2, NULL); }

if_block:
  IF `(' condition `)'
  `{' statement_list `}'
  { $$ = new IfNode($3, $5); }
| IF `(' condition `)' `;'
  { $$ = new IfNode($3, Node()); }

value_list:
  %empty { $$ = new vector<Val>(); }
| value_list value
  { $1->append($2); $$ = $1; }

stmt_list:
  %empty { $$ = new vector<St>(); }
| stmt_list stmt
  { $1->append($2); $$ = $1; }
\end{lstlisting}
\tcblower%
  \begin{lstlisting}
enum:
  ENUM name=NAME
  block=opt_block(list(value))
  { EnumNode (name, block) }

if_block:
  IF `(' cond=condition `)'
  block=opt_block(list(stmt))
  { let inside = match block with
    | None -> EmptyNode
    | Some a -> Node a
    in IfNode inside }

opt_block(INSIDE):
| `{' res=INSIDE `}'
  { Some res }
| `;'    { None }
  \end{lstlisting}
  \end{tcolorbox}
\end{lrbox}
